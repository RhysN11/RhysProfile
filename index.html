// app.js import express from "express"; import fetch from "node-fetch"; import cors from "cors"; import path from "path"; import { fileURLToPath } from "url"; const app = express(); app.use(cors()); const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); // ------------------------- // ðŸ”‘ API KEYS (set as env vars in Render/Vercel) // ------------------------- const EBAY_CLIENT_ID = process.env.EBAY_CLIENT_ID || "YOUR_EBAY_CLIENT_ID"; const EBAY_CLIENT_SECRET = process.env.EBAY_CLIENT_SECRET || "YOUR_EBAY_CLIENT_SECRET"; const ETSY_API_KEY = process.env.ETSY_API_KEY || "YOUR_ETSY_API_KEY"; // ------------------------- // eBay Token Fetch // ------------------------- async function getEbayToken() { const resp = await fetch("https://api.ebay.com/identity/v1/oauth2/token", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded", Authorization: "Basic " + Buffer.from(`${EBAY_CLIENT_ID}:${EBAY_CLIENT_SECRET}`).toString("base64"), }, body: "grant_type=client_credentials&scope=https://api.ebay.com/oauth/api_scope", }); const json = await resp.json(); return json.access_token; } // ------------------------- // eBay Search by Keyword // ------------------------- async function searchEbay(q) { const token = await getEbayToken(); const resp = await fetch( `https://api.ebay.com/buy/browse/v1/item_summary/search?q=${encodeURIComponent( q )}&limit=5`, { headers: { Authorization: `Bearer ${token}` } } ); const data = await resp.json(); return (data.itemSummaries || []).map((item) => ({ title: item.title, price: item.price?.value, currency: item.price?.currency, url: item.itemWebUrl, source: "eBay", })); } // ------------------------- // Etsy Search // ------------------------- async function searchEtsy(q) { const resp = await fetch( `https://openapi.etsy.com/v3/application/listings/active?limit=5&keywords=${encodeURIComponent( q )}`, { headers: { "x-api-key": ETSY_API_KEY } } ); const data = await resp.json(); return (data.results || []).map((item) => ({ title: item.title, price: item.price?.amount / 100, currency: item.price?.currency_code, url: `https://www.etsy.com/listing/${item.listing_id}`, source: "Etsy", })); } // ------------------------- // API Route // ------------------------- app.get("/api/search", async (req, res) => { const q = req.query.q || "shoes"; try { const [ebay, etsy] = await Promise.all([searchEbay(q), searchEtsy(q)]);
