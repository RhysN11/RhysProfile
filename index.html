// app.js import express from "express"; import fetch from "node-fetch"; import cors from "cors"; import multer from "multer"; import fs from "fs"; const app = express(); app.use(cors()); const upload = multer({ dest: "uploads/" }); // ðŸ”‘ API Keys (set these in Render/Vercel environment settings) const EBAY_CLIENT_ID = process.env.EBAY_CLIENT_ID || "YOUR_EBAY_CLIENT_ID"; const EBAY_CLIENT_SECRET = process.env.EBAY_CLIENT_SECRET || "YOUR_EBAY_CLIENT_SECRET"; const ETSY_API_KEY = process.env.ETSY_API_KEY || "YOUR_ETSY_API_KEY"; // ------------------------- // eBay Auth // ------------------------- async function getEbayToken() { const resp = await fetch("https://api.ebay.com/identity/v1/oauth2/token", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded", Authorization: "Basic " + Buffer.from(`${EBAY_CLIENT_ID}:${EBAY_CLIENT_SECRET}`).toString("base64"), }, body: "grant_type=client_credentials&scope=https://api.ebay.com/oauth/api_scope", }); const json = await resp.json(); return json.access_token; } // ------------------------- // eBay Search by Image // ------------------------- app.post("/api/searchByImage", upload.single("photo"), async (req, res) => { try { const token = await getEbayToken(); const imgBuffer = fs.readFileSync(req.file.path); fs.unlinkSync(req.file.path); // cleanup const resp = await fetch("https://api.ebay.com/buy/browse/v1/item_summary/search_by_image?limit=5", { method: "POST", headers: { "Authorization": `Bearer ${token}`, "Content-Type": "image/jpeg", }, body: imgBuffer }); const data = await resp.json(); const items = (data.itemSummaries || []).map((item) => ({ title: item.title, price: item.price?.value, currency: item.price?.currency, url: item.itemWebUrl, source: "eBay (Image)" })); res.json({ items }); } catch (err) { console.error(err); res.status(500).json({ error: "Image search failed" }); } }); // ------------------------- // Etsy Search by Keyword (optional extra) // ------------------------- app.get("/api/searchEtsy", async (req, res) => { const q = req.query.q || "shoes"; try { const resp = await fetch( `https://openapi.etsy.com/v3/application/listings/active?limit=5&keywords=${encodeURIComponent(q)}`, { headers: { "x-api-key": ETSY_API_KEY } } ); const data = await resp.json(); const items = (data.results || []).map((item) => ({ title: item.title, price: item.price?.amount / 100, currency: item.price?.currency_code, url: `https://www.etsy.com/listing/${item.listing_id}`, source: "Etsy" })); res.json({ items }); } catch (err) { console.error(err); res.status(500).json({ error: "Etsy search failed" }); } }); // ------------------------- // Frontend Website // ------------------------- app.get("/", (req, res) => { res.send(` Rhys Newman | Visual Finder

`); }); // ------------------------- // Run Server // ------------------------- const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(\`âœ… App running on http://localhost:\${PORT}\`));
